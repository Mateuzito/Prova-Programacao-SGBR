Função para somar elementos do array: : [0.5, 7, 2, 35, 3.5, 4, 40, 24]

var array = [0.5, 7, 2, 35, 3.5, 4, 40, 24]; //inserindo os elementos a serem somados na variável
var soma = array.reduce(function(soma, i) { 
    return soma + i; //somatório usando o reduce
});
console.log(soma);
